Algorithm GaussianEliminationTotalPivoting(A, b, n)

Initialize marks array to keep track of variable permutations: marks = [1, 2, ..., n]

// Phase 1: Forward Elimination with Total Pivoting
For k = 1 to n-1 do
    // Find absolute maximum in the submatrix A[k..n][k..n]
    max_val = 0
    max_row = k
    max_col = k
    For r = k to n do
        For c = k to n do
            If |A[r][c]| > max_val then
                max_val = |A[r][c]|
                max_row = r
                max_col = c
            EndIf
        EndFor
    EndFor
    
    If max_val == 0 then
        Print "System has no unique solution"
        Stop
    EndIf
    
    If max_row != k then
        Swap row max_row with row k in A
        Swap b[max_row] with b[k]
    EndIf
    
    If max_col != k then
        Swap column max_col with column k in A
        Swap marks[max_col] with marks[k]
    EndIf
    
    // Elimination
    For i = k+1 to n do
        multiplier = A[i][k] / A[k][k]
        For j = k to n do
            A[i][j] = A[i][j] - multiplier * A[k][j]
        EndFor
        b[i] = b[i] - multiplier * b[k]
    EndFor
EndFor

// Phase 2: Back Substitution
Initialize x vector of size n
x[n] = b[n] / A[n][n]

For i = n-1 down to 1 do
    sum = 0
    For j = i+1 to n do
        sum = sum + A[i][j] * x[j]
    EndFor
    x[i] = (b[i] - sum) / A[i][i]
EndFor

// Phase 3: Un-permute the variables based on marks
Initialize final_x vector of size n
For i = 1 to n do
    final_x[marks[i]] = x[i]
EndFor

Print final_x
